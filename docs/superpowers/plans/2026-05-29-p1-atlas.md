```ts
import { readFile } from 'node:fs/promises';
import { join } from 'node:path';
import type { KnowledgeStore } from '../storage/store.js';
import type { KnowledgeAtom, AtomRelationRow } from '../types/atoms.js';
import type { StoredKnowledge, KnowledgeGap } from '../types/knowledge.js';
import type { SourceFileRecord } from '../source-sync/types.js';
import { buildAreaModel, deriveAreaKey, type ProjectArea } from '../knowledge-areas/area-model.js';

export interface AreaDepEdge { from: string; to: string; weight: number }

export interface AtlasInputs {
  project: string;
  repoPath: string;
  generatedAt: string;
  areas: ProjectArea[];
  atoms: KnowledgeAtom[];
  knowledge: StoredKnowledge[];
  relations: AtomRelationRow[];
  ledger: SourceFileRecord[];
  knowledgeGaps: KnowledgeGap[];
  openConflictCount: number;
  scripts: Record<string, string>;
  readmeCommands?: string;
  areaDeps: AreaDepEdge[];
}

/** Best-effort file path for an atom: first trigger file, else first file evidence. */
function atomPath(atom: KnowledgeAtom): string | undefined {
  const t = atom.trigger.files?.find((f) => f.length > 0);
  if (t) return t;
  for (const ev of atom.evidence) if (ev.kind === 'file' && ev.path) return ev.path;
  return undefined;
}

function buildAreaDeps(atoms: KnowledgeAtom[], relations: AtomRelationRow[]): AreaDepEdge[] {
  const areaOf = new Map<string, string>();
  for (const atom of atoms) {
    const p = atomPath(atom);
    areaOf.set(atom.id, p ? deriveAreaKey(p) : '_unassigned');
  }
  const weights = new Map<string, number>();
  for (const rel of relations) {
    if (rel.relationType !== 'depends_on' && rel.relationType !== 'refines') continue;
    const from = areaOf.get(rel.fromAtomId);
    const to = areaOf.get(rel.targetAtomId);
    if (!from || !to || from === to) continue;
    const key = `${from} ${to}`;
    weights.set(key, (weights.get(key) ?? 0) + 1);
  }
  return [...weights.entries()]
    .map(([k, weight]) => { const [from, to] = k.split(' '); return { from, to, weight }; })
    .sort((a, b) => (a.from === b.from ? a.to.localeCompare(b.to) : a.from.localeCompare(b.from)));
}

async function readScripts(repoPath: string): Promise<Record<string, string>> {
  try {
    const raw = await readFile(join(repoPath, 'package.json'), 'utf8');
    const pkg = JSON.parse(raw) as { scripts?: Record<string, string> };
    return pkg.scripts ?? {};
  } catch { return {}; }
}

async function readReadmeCommands(repoPath: string): Promise<string | undefined> {
  try {
    const raw = await readFile(join(repoPath, 'README.md'), 'utf8');
    const m = raw.match(/^#+\s*Commands\b[\s\S]*?(?=\n#+\s|\n*$)/im);
    return m ? m[0].trim() : undefined;
  } catch { return undefined; }
}

export async function gatherAtlasInputs(
  store: KnowledgeStore,
  opts: { project: string; repoPath: string; generatedAt: string },
): Promise<AtlasInputs> {
  const { project, repoPath, generatedAt } = opts;
  const [areas, atoms, knowledge, relations, ledger, knowledgeGaps, conflicts] = await Promise.all([
    buildAreaModel(store, project),
    store.listAtoms({ project, limit: 100_000 }),
    store.listKnowledge({ project, limit: 100_000 }),
    store.listAtomRelations({ project, limit: 1_000_000 }),
    store.listSourceFiles({ project, limit: 100_000 }),
    store.listKnowledgeGaps({ project, limit: 100_000 }),
    store.listAtomImportConflicts({ project, status: 'open', limit: 100_000 }),
  ]);
  const [scripts, readmeCommands] = await Promise.all([readScripts(repoPath), readReadmeCommands(repoPath)]);
  return {
    project, repoPath, generatedAt, areas, atoms, knowledge, relations, ledger, knowledgeGaps,
    openConflictCount: conflicts.length, scripts, readmeCommands,
    areaDeps: buildAreaDeps(atoms, relations),
  };
}
```

```ts
import { basename } from 'node:path';
import type { AtlasInputs } from './inputs.js';
import type { KnowledgeAtom } from '../types/atoms.js';
import type { StoredKnowledge } from '../types/knowledge.js';

const NAV = '<!-- atlas: project-map.md · flows.md · commands.md · risks.md · open-gaps.md -->';

export function atlasHeader(inputs: AtlasInputs, hash8: string, title: string): string {
  return `<!-- atlas_run input ${hash8} · generated ${inputs.generatedAt} -->\n${NAV}\n\n# ${title}\n`;
}

function topN<T>(items: T[], n: number, score: (t: T) => string): T[] {
  return [...items].sort((a, b) => score(a).localeCompare(score(b))).slice(0, n);
}

function purposeLine(area: AtlasInputs['areas'][number], inputs: AtlasInputs): string {
  const inArea = new Set(area.knowledgeIds);
  const docs = inputs.knowledge
    .filter((k) => inArea.has(k.id) && (k.itemType === 'wiki' || k.itemType === 'spec'))
    .sort((a, b) => b.trustLevel - a.trustLevel);
  const top = docs[0];
  if (top) {
    const text = (top.summary && top.summary.trim()) || top.content.split(/(?<=\.)\s/)[0] || '';
    if (text.trim()) return text.trim();
  }
  if (area.labels.length > 0) {
    return `labels: ${area.labels.map((l) => `${l.type}/${l.value}`).join(', ')}`;
  }
  return '(no description — see open-gaps.md)';
}

function keyFiles(area: AtlasInputs['areas'][number], inputs: AtlasInputs): string[] {
  const counts = new Map<string, number>();
  for (const p of area.paths) counts.set(p, 0);
  const bump = (p: string | undefined) => { if (p && counts.has(p)) counts.set(p, (counts.get(p) ?? 0) + 1); };
  for (const k of inputs.knowledge) bump((k.metadata as { sourcePath?: string }).sourcePath);
  for (const a of inputs.atoms) bump(a.trigger.files?.[0]);
  return [...counts.entries()]
    .sort((a, b) => (b[1] - a[1]) || a[0].localeCompare(b[0]))
    .slice(0, 5)
    .map(([p]) => basename(p));
}

function keySymbols(area: AtlasInputs['areas'][number], inputs: AtlasInputs): string[] {
  const inArea = new Set(area.atomIds);
  const freq = new Map<string, number>();
  for (const a of inputs.atoms) {
    if (!inArea.has(a.id)) continue;
    for (const s of a.trigger.symbols ?? []) freq.set(s, (freq.get(s) ?? 0) + 1);
  }
  for (const k of inputs.knowledge) {
    if (!area.knowledgeIds.includes(k.id)) continue;
    for (const l of k.labels) if (l.type === 'symbol') freq.set(l.value, (freq.get(l.value) ?? 0) + 1);
  }
  return [...freq.entries()].sort((a, b) => (b[1] - a[1]) || a[0].localeCompare(b[0])).slice(0, 8).map(([s]) => s);
}

export function buildProjectMap(inputs: AtlasInputs): string {
  const lines: string[] = ['Entry point for understanding this project. Generated by `tuberosa atlas`.\n'];
  for (const area of inputs.areas) {
    const purpose = purposeLine(area, inputs);
    const files = keyFiles(area, inputs);
    const symbols = keySymbols(area, inputs);
    const deps = inputs.areaDeps.filter((e) => e.from === area.key).map((e) => e.to);
    lines.push(`## ${area.key} — ${area.label}`);
    lines.push(purpose);
    if (files.length) lines.push(`key files:   ${files.join(', ')}`);
    if (symbols.length) lines.push(`key symbols: ${symbols.join(', ')}`);
    lines.push(
      `files ${area.counts.files} · knowledge ${area.counts.knowledge} · atoms ${area.counts.atoms} ` +
      `(${area.counts.verifiedAtoms} verified) · crossing edges ${area.crossingRelations}`,
    );
    if (deps.length) lines.push(`→ depends on: ${deps.join(', ')}`);
    lines.push('');
  }
  return `${atlasHeader(inputs, 'PENDING', 'Project Map')}\n${lines.join('\n')}`;
}
```

```ts
export function buildFlows(inputs: AtlasInputs): string {
  const lines: string[] = ['_Best-effort, derived from Tuberosa\'s own knowledge graph. See open-gaps.md._\n'];
  lines.push('## Area dependency map');
  if (inputs.areaDeps.length === 0) {
    lines.push('No cross-area dependencies inferred yet.');
  } else {
    for (const e of inputs.areaDeps) lines.push(`- ${e.from} → ${e.to} (${e.weight})`);
  }
  lines.push('', '## Co-change coupling');
  const coChange = inputs.relations.filter((r) => r.relationType === 'co_changes_with');
  if (coChange.length === 0) {
    lines.push('No co-change data — run `pnpm run infer-co-change` to populate temporal coupling.');
  } else {
    lines.push(`${coChange.length} co-change relations recorded.`);
  }
  return `${atlasHeader(inputs, 'PENDING', 'Flows')}\n${lines.join('\n')}`;
}
```

```ts
const COMMAND_GROUPS: { title: string; match: (n: string) => boolean }[] = [
  { title: 'Build & Dev', match: (n) => /^(build|dev|start|workbench)/.test(n) },
  { title: 'Test & Eval', match: (n) => /^(test|eval:|sandbox|benchmark|calibrate-fusion)/.test(n) },
  { title: 'Data & Maintenance', match: (n) => /^(backfill|archival-sweep|infer-co-change|prune-stale-edges|cluster-|migrate|seed|import:docs|backup|restore)/.test(n) },
  { title: 'Packs', match: (n) => /^(export-pack|import-pack)$/.test(n) },
  { title: 'Ops', match: (n) => /^(mcp|worker|error-logs|organization|context-quality)$/.test(n) },
];

export function buildCommands(inputs: AtlasInputs): string {
  const names = Object.keys(inputs.scripts).sort();
  const used = new Set<string>();
  const lines: string[] = [];
  for (const group of COMMAND_GROUPS) {
    const members = names.filter((n) => !used.has(n) && group.match(n));
    if (members.length === 0) continue;
    members.forEach((n) => used.add(n));
    lines.push(`### ${group.title}`);
    for (const n of members) lines.push(`- \`pnpm run ${n}\` — \`${inputs.scripts[n]}\``);
    lines.push('');
  }
  const other = names.filter((n) => !used.has(n));
  if (other.length) {
    lines.push('### Other');
    for (const n of other) lines.push(`- \`pnpm run ${n}\` — \`${inputs.scripts[n]}\``);
    lines.push('');
  }
  if (inputs.readmeCommands) {
```

```ts
export function buildRisks(inputs: AtlasInputs): string {
  const lines: string[] = [];

  const gotchas = inputs.atoms.filter((a) => a.type === 'gotcha')
    .sort((a, b) => a.tier.localeCompare(b.tier) || a.claim.localeCompare(b.claim));
  lines.push('## Gotchas');
  if (gotchas.length === 0) lines.push('None recorded.');
  for (const g of gotchas) {
    lines.push(`- ${g.claim}`);
    for (const p of g.pitfalls ?? []) lines.push(`  - pitfall: ${p}`);
  }

  const staleStatuses = new Set(['changed', 'missing']);
  const stalePaths = new Set(inputs.ledger.filter((f) => staleStatuses.has(f.status)).map((f) => f.path));
  const stale = inputs.knowledge.filter((k) => {
    const p = (k.metadata as { sourcePath?: string }).sourcePath;
    return p ? stalePaths.has(p) : false;
  });
  lines.push('', '## Stale knowledge');
  if (stale.length === 0) lines.push('None — all knowledge maps to tracked files.');
  for (const k of stale) lines.push(`- ${(k.metadata as { sourcePath?: string }).sourcePath} — ${k.title}`);

  lines.push('', '## Open conflicts');
  lines.push(`${inputs.openConflictCount} open import conflict(s).`);

  return `${atlasHeader(inputs, 'PENDING', 'Risks')}\n${lines.join('\n')}`;
}
```

```ts
import { mkdir, writeFile } from 'node:fs/promises';
import { join } from 'node:path';
import type { KnowledgeStore, AtlasRunRecord } from '../storage/store.js';
import { gatherAtlasInputs, type AtlasInputs } from './inputs.js';
import { buildProjectMap, buildFlows, buildCommands, buildRisks, buildOpenGaps } from './builders.js';
import { sha256OfBuffer } from '../export/manifest.js';

export interface AtlasRegenArgs { project: string; repoPath: string; generatedAt: string; write?: boolean }
export interface AtlasRegenResult {
  inputHash: string;
  files: { name: string; bytes: number }[];
  contents: { name: string; content: string }[];
  atlasRun?: AtlasRunRecord;
}

const BUILDERS: { name: string; build: (i: AtlasInputs) => string }[] = [
  { name: 'project-map.md', build: buildProjectMap },
  { name: 'flows.md', build: buildFlows },
  { name: 'commands.md', build: buildCommands },
  { name: 'risks.md', build: buildRisks },
  { name: 'open-gaps.md', build: buildOpenGaps },
];

export class AtlasService {
  constructor(private store: KnowledgeStore, private config: { atlasDir: string }) {}

  async regenerate(args: AtlasRegenArgs): Promise<AtlasRegenResult> {
    const inputs = await gatherAtlasInputs(this.store, args);
    // Hash covers inputs only (NOT generatedAt) so an unchanged project is stable.
    const hashable = JSON.stringify({ ...inputs, generatedAt: undefined, repoPath: undefined });
    const inputHash = sha256OfBuffer(hashable);
    const hash8 = inputHash.slice(0, 8);

    const contents = BUILDERS.map(({ name, build }) => ({
      name,
      content: build(inputs).replace(/input PENDING/g, `input ${hash8}`),
    }));
    const files = contents.map((c) => ({ name: c.name, bytes: Buffer.byteLength(c.content, 'utf8') }));

    let atlasRun: AtlasRunRecord | undefined;
    if (args.write) {
      await mkdir(this.config.atlasDir, { recursive: true });
      for (const c of contents) await writeFile(join(this.config.atlasDir, c.name), c.content, 'utf8');
      atlasRun = await this.store.createAtlasRun({ project: args.project, inputHash, files, generatedAt: args.generatedAt });
    }
    return { inputHash, files, contents, atlasRun };
  }
}
```

```ts
export interface AtlasCommandDeps {

export async function atlasCommand(
  invocation: CliInvocation,
  io: CommandIo,
  deps: AtlasCommandDeps,
): Promise<CommandResult> {
  const project = typeof invocation.options.project === 'string' ? invocation.options.project : '';
  if (!project) {
    io.err('tuberosa atlas requires --project <name>');
    return { exitCode: 1 };
  }
  const repoPath = typeof invocation.options.path === 'string' ? invocation.options.path : io.cwd;
  const write = invocation.options.write === true;
  const asJson = invocation.options.json === true;

  const service = await deps.makeService(project, repoPath);
  const generatedAt = new Date().toISOString();
  const result = await service.regenerate({ project, repoPath, generatedAt, write });

  if (asJson) {

    for (const f of result.files) io.out(`  ${f.name} — ${f.bytes} bytes`);
    if (!write) io.out('Re-run with --write to persist to .tuberosa/atlas/.');
  }
  return { exitCode: 0 };
}
```
